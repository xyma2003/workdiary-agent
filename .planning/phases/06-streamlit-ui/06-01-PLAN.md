        repo_path = st.text_input(

            placeholder="/path/to/your/repo",
            key="repo_path_field",
        )
        data_input = st.text_area(
            "数据/指标（可选粘贴）",
            placeholder="DAU: 12000, 转化率: 3.2%, ...",

            key="data_input_field",

    if submitted:
        if not raw_input.strip():
